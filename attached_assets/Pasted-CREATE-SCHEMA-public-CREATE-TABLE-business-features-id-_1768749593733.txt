CREATE SCHEMA "public";
CREATE TABLE "business_features" (
	"id" uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
	"feature_name" varchar(255),
	"benefit_statement" text,
	"linked_pain_point_id" uuid
);
CREATE TABLE "client_segments" (
	"id" uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
	"segment_name" varchar(255) NOT NULL,
	"job_titles" jsonb,
	"psychographics" text,
	"language_style" jsonb,
	"created_at" timestamp with time zone DEFAULT now()
);
CREATE TABLE "execution_log" (
	"id" uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
	"strategy_id" uuid,
	"post_text" text,
	"linkedin_post_url" text,
	"published_at" timestamp with time zone DEFAULT now(),
	"analyzed" boolean DEFAULT false,
	"views" integer DEFAULT 0,
	"likes" integer DEFAULT 0,
	"comments" integer DEFAULT 0
);
CREATE TABLE "pain_points" (
	"id" uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
	"description" text NOT NULL,
	"segment_id" uuid,
	"severity_score" integer,
	"created_at" timestamp with time zone DEFAULT now()
);
CREATE TABLE "strategy_performance_ledger" (
	"strategy_id" uuid PRIMARY KEY DEFAULT uuid_generate_v4(),
	"segment_id" uuid,
	"post_type" varchar(50),
	"alpha" double precision DEFAULT 1.0,
	"beta" double precision DEFAULT 1.0,
	"last_updated" timestamp with time zone DEFAULT now()
);
ALTER TABLE "business_features" ADD CONSTRAINT "business_features_linked_pain_point_id_fkey" FOREIGN KEY ("linked_pain_point_id") REFERENCES "pain_points"("id");
ALTER TABLE "execution_log" ADD CONSTRAINT "execution_log_strategy_id_fkey" FOREIGN KEY ("strategy_id") REFERENCES "strategy_performance_ledger"("strategy_id");
ALTER TABLE "pain_points" ADD CONSTRAINT "pain_points_segment_id_fkey" FOREIGN KEY ("segment_id") REFERENCES "client_segments"("id");
ALTER TABLE "strategy_performance_ledger" ADD CONSTRAINT "strategy_performance_ledger_segment_id_fkey" FOREIGN KEY ("segment_id") REFERENCES "client_segments"("id");
CREATE UNIQUE INDEX "business_features_pkey" ON "business_features" ("id");
CREATE UNIQUE INDEX "client_segments_pkey" ON "client_segments" ("id");
CREATE UNIQUE INDEX "execution_log_pkey" ON "execution_log" ("id");
CREATE UNIQUE INDEX "pain_points_pkey" ON "pain_points" ("id");
CREATE UNIQUE INDEX "strategy_performance_ledger_pkey" ON "strategy_performance_ledger" ("strategy_id");
