interface PasswordCreationData {

  error?: string;

    password: '',

    const isValid = validateForm();

    const passwordsMatch = formData.password === formData.confirmPassword;

    setFormData(prev => ({
      ...prev,
      password: value,
      error: undefined
    }));

    setFormData(prev => ({
      ...prev,
      confirmPassword: value,
      error: undefined

    setTouched(prev => ({ ...prev, [field]: true }));
